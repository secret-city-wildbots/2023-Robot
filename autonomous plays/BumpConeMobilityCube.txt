This play scores one cone mid or high on the bump side, and then picks up a cube and scores it mid or high.
Supports dashboard scoring position selection for second piece only
Does not support dashboard state selection
*
start_x_red = 74
start_y_red = 299
start_h_red = 0
start_x_blue = 74
start_y_blue = 17
start_h_blue = 0
*
DriveWaitForManip(8)
DrivePath(6,10,-1,Cone1ToCube2Bump,robot_suffix) # drive to cube 2
DrivePath(6,10,-1,Cube2ToScoreBump,robot_suffix) # drive to score cube 2
DriveWaitForManip(24)
DriveAutoTarget(on)
DriveWaitForManip(29)
DrivePath(6,10,-1,ReadyForTeleOpBump,robot_suffix)
DriveEnd()
*
ManipChangePiece(cone)
ManipChangeScoringPosition(0)
ManipChangeState(2)
ManipWaitForArm(stationary)
ManipDropPiece()				# score cone 1
ManipChangePiece(cube)
ManipChangeState(4)
ManipWaitForArm(safe)
ManipWaitForDrive(1)
ManipWaitForDistance(45,-1)
ManipSetRobotPosition(-1,-1)
ManipWaitForDistance(-144,-1)
ManipGroundPickup(on)			# pickup cube 2
ManipWaitForDrive(2)
ManipGroundPickup(off)
ManipChangeState(0)
ManipWaitForDistance(-84, -1)
ManipChangeScoringPosition(-1)
ManipWaitForArm(stationary)
ManipChangeState(2)
ManipWaitForDistance(-18, -1)
ManipSetRobotPosition(-1,-1)
ManipWaitForDrive(3)
ManipSetRobotPosition(-1,-1)
ManipWaitForDrive(5)
ManipWaitForArm(stationary)
ManipDropPiece()				# score cube 2
ManipChangeState(0)
ManipWaitForArm(safe)
ManipChangeState(-1)
ManipEnd()
