start_y_red = 141
start_h_red = 0
start_x_blue = 74
start_y_blue = 175

*

DrivePath(6,10,-1,Cube2ToScoreLowLeft,robot_suffix) # drive to score cube 2
DriveWaitForManip(17)
DriveBalance(near)
DriveEnd()
*
ManipChangePiece(cone)
ManipChangeState(0)
ManipGroundScore() 				# score cone 1
ManipChangePiece(cube)
ManipChangeState(4)
ManipWaitForDrive(1)
ManipWaitForDistance(1,-1)
ManipGroundPickup(on) 		
ManipWaitForDistance(53,-1)
ManipSetRobotPosition(-1,-1)	# pickup cube 2
ManipWaitForDrive(2)
ManipGroundPickup(off)

ManipWaitForDistance(-18, -1)
ManipSetRobotPosition(-1,-1)
ManipWaitForDrive(3)
ManipGroundScore() 				# score cube 2
ManipChangeState(1)
ManipEnd()
