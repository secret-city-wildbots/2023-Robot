This play scores one game piece high, goes over charging station to get the mobility bonus, then balances.

*
start_x_red = 75
start_y_red = 200
start_h_red = 0
start_x_blue = 75
start_y_blue = 130
start_h_blue = 0
*
DriveWaitForManip(7)
DriveOver(near)
DriveBalance(far)
DriveEnd()
*
ManipChangePiece(cone)
ManipChangeScoringPosition(-1)
ManipChangeState(2)
ManipWaitForArm(stationary)
ManipDropPiece()
ManipChangeState(1)
ManipWaitForArm(safe)
ManipEnd()
